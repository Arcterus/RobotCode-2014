006package RobotCode2014.commands.manipulator;
007
008import RobotCode2014.Constants;
009import RobotCode2014.commands.CommandBase;
010import edu.wpi.first.wpilibj.command.CommandGroup;
011import edu.wpi.first.wpilibj.command.WaitCommand;
012
013/**
014 *
015 * @author Manan Shah
016 * @author Ben H. Fires the ball (assumes that the winch is already wound up and
017 * clamp is open) and retracts the piston again.
018 */
019public class LaunchBallCommand extends CommandGroup {
020
021    public LaunchBallCommand() {
022        addSequential(new OpenClampCommand());
023        addSequential(new WaitCommand(Constants.ManipulatorConst.launchAfterOpenClampDelay));
024        addSequential(new UnlockPistonCommand());
025        addSequential(new WaitCommand(Constants.ManipulatorConst.retractAfterLaunchDelay));
026
027        addParallel(new CloseClampCommand());
028        addSequential(new LockAndRetractPistonCommand());
029    }
030
031}
