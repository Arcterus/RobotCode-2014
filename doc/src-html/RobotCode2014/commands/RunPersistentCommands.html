001package RobotCode2014.commands;
002
003import RobotCode2014.commands.drivetrain.DrivetrainAccelerationCommand;
004import RobotCode2014.commands.manipulator.PitchAccelerationCommand;
005import RobotCode2014.commands.misc.UpdateEncodersCommand;
006import RobotCode2014.commands.misc.DebugCommand;
007import edu.wpi.first.wpilibj.command.CommandGroup;
008
009/**
010 * A command group that starts all commands that should run continuously on the robot.
011 * (Exists mainly just to save space in the scheduler display on the SmartDashboard.)
012 * @author Brian Chan
013 */
014public class RunPersistentCommands extends CommandGroup {
015    
016    public RunPersistentCommands() {
017        addParallel(new DebugCommand());
018        addParallel(new UpdateEncodersCommand());
019        addParallel(new DrivetrainAccelerationCommand());
020        addParallel(new PitchAccelerationCommand());
021    }
022    
023}
