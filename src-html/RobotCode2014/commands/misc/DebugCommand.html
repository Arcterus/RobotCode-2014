002package RobotCode2014.commands.misc;
003
004import RobotCode2014.commands.CommandBase;
005import RobotCode2014.wrappers.EncoderWrapper;
006import edu.wpi.first.wpilibj.Encoder;
007import edu.wpi.first.wpilibj.smartdashboard.SmartDashboard;
008
009/**
010 * A command that constantly runs, writing debug information to SmartDashboard
011 * @author Manan
012 */
013public class DebugCommand extends CommandBase {
014    
015    Encoder leftEnc, rightEnc;
016    EncoderWrapper pitchEnc;
017    public DebugCommand() {
018        setRunWhenDisabled(true);
019    }
020
021    // Called just before this Command runs the first time
022    protected void initialize() {
023        leftEnc = drivetrain.getLeftEnc();
024        rightEnc = drivetrain.getRightEnc();
025    }
026
027    // Called repeatedly when this Command is scheduled to run
028    protected void execute() {
029        SmartDashboard.putNumber("Left rate: ", leftEnc.getRate());
030        SmartDashboard.putNumber("Right rate: ", rightEnc.getRate());
031        SmartDashboard.putNumber("Left raw: ", leftEnc.getRaw());
032        SmartDashboard.putNumber("Right raw: ", rightEnc.getRaw());
033        SmartDashboard.putNumber("Left ticks: ", leftEnc.get());
034        SmartDashboard.putNumber("Right ticks: ", rightEnc.get());
035        SmartDashboard.putNumber("Left dist: ", leftEnc.getDistance());
036        SmartDashboard.putNumber("Right dist: ", rightEnc.getDistance());
037        
038        SmartDashboard.putBoolean("piston ls ", manipPiston.isPistonRetracted());
039        SmartDashboard.putBoolean("pitch l", manipPitch.isMinPitch());
040        SmartDashboard.putBoolean("pressure ", pneumatics.getPressureSwitch());
041        
042        SmartDashboard.putNumber("pitch ", manipPitch.getPitchRaw());   
043        SmartDashboard.putNumber("pitch ", manipPitch.getPitchRads());  
044        
045        SmartDashboard.putBoolean("mag1 ", manipPitch.source1());
046        SmartDashboard.putBoolean("mag2 ", manipPitch.source2());
047        
048        SmartDashboard.putBoolean("goal hot", networking.isGoalHot());
049        SmartDashboard.putNumber("goal angle", networking.getGoalAngle());
050        SmartDashboard.putNumber("goal dist", networking.getGoalDistance());
051        
052        
053        //SmartDashboard.putBoolean("pid enabled",manipPitch.getPIDController().isEnable());
054        //SmartDashboard.putNumber("pid enabled",manipulator.getPIDController().);
055        
056    }
057
058    // Make this return true when this Command no longer needs to run execute()
059    protected boolean isFinished() {
060        return false;
061    }
062
063    // Called once after isFinished returns true
064    protected void end() {
065    }
066
067    // Called when another command which requires one or more of the same
068    // subsystems is scheduled to run
069    protected void interrupted() {
070    }
071}
