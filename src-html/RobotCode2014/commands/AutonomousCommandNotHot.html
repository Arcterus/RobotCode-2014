013public class AutonomousCommandNotHot extends CommandGroup {
014    
015    public AutonomousCommandNotHot(double angle, double time, double left, double right) {
016        //sets the angle to 60 degrees
017            System.out.println("Started commmand chain.");
018            addParallel(new PitchToAngleCommand(angle));
019            System.out.println("Finished MoveArmToPitch");
020            addSequential(new DriveForTimeCommand(time, left, right));
021            System.out.println("Finished DriveForTime");
022            addSequential(new WaitForChildren());
023            System.out.println("Waited for Chillun");
024            //addSequential(new LaunchBallCommand());
025            System.out.println("Launched ball, yay!");
026    }
027}
