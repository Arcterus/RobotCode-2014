001package RobotCode2014;
002
003import edu.wpi.first.wpilibj.command.Command;
004import edu.wpi.first.wpilibj.command.Scheduler;
005import edu.wpi.first.wpilibj.smartdashboard.SmartDashboard;
006
007import RobotCode2014.commands.CommandBase;
008import RobotCode2014.commands.drivetrain.*;
009import RobotCode2014.commands.manipulator.*;
010import RobotCode2014.commands.misc.*;
011import RobotCode2014.wrappers.GamePadWrapper;
012import RobotCode2014.wrappers.JoystickButtonWrapper;
013
014
015/**
016 * This class is the glue that binds the controls on the physical operator
017 * interface to the commands and command groups that allow control of the robot.
018 */
019public class OI {
020    
021    public static Command manualDriveRestore = new RestoreManualDriveCommand();
022    public static Command manualDrive = new ManualDriveCommand();
023    
024    public static Command manualPitch = new ManualPitchAndRetractCommand();
025    public static Command manualPitchRestore = new RestoreManualPitchCommand();
026    
027    public static GamePadWrapper gamepad1 = new GamePadWrapper(1);
028    public static GamePadWrapper gamepad2 = new GamePadWrapper(2);
029    
030    public OI() {
031        SmartDashboard.putData(Scheduler.getInstance());
032        SmartDashboard.putData(CommandBase.drivetrain);
033        SmartDashboard.putData(CommandBase.manipPitch);
034        
035
036        SmartDashboard.putData(new UpdateConstantsCommand());
037        
038        /**************Button mapping*******************
039         * The below commands map buttons to the commands that are supposed to 
040         *  operate when the buttons are pressed.
041         *  
042         * @author Manan
043         * @author Andrew
044         * @author Vedaad
045         ***********************************************/
046        JoystickButtonWrapper g1b1 = new RobotCode2014.wrappers.JoystickButtonWrapper(gamepad1, 1);
047        JoystickButtonWrapper g1b2 = new RobotCode2014.wrappers.JoystickButtonWrapper(gamepad1, 2);
048        JoystickButtonWrapper g1b3 = new RobotCode2014.wrappers.JoystickButtonWrapper(gamepad1, 3);
049        JoystickButtonWrapper g1b4 = new RobotCode2014.wrappers.JoystickButtonWrapper(gamepad1, 4);
050        JoystickButtonWrapper g1b5 = new RobotCode2014.wrappers.JoystickButtonWrapper(gamepad1, 5);
051        JoystickButtonWrapper g1b6 = new RobotCode2014.wrappers.JoystickButtonWrapper(gamepad1, 6);
052        JoystickButtonWrapper g1b7 = new RobotCode2014.wrappers.JoystickButtonWrapper(gamepad1, 7);
053        JoystickButtonWrapper g1b8 = new RobotCode2014.wrappers.JoystickButtonWrapper(gamepad1, 8);
054        
055        JoystickButtonWrapper g2b1 = new RobotCode2014.wrappers.JoystickButtonWrapper(gamepad2, 1);
056        JoystickButtonWrapper g2b2 = new RobotCode2014.wrappers.JoystickButtonWrapper(gamepad2, 2);
057        JoystickButtonWrapper g2b3 = new RobotCode2014.wrappers.JoystickButtonWrapper(gamepad2, 3);
058        JoystickButtonWrapper g2b4 = new RobotCode2014.wrappers.JoystickButtonWrapper(gamepad2, 4);
059        JoystickButtonWrapper g2b5 = new RobotCode2014.wrappers.JoystickButtonWrapper(gamepad2, 5);
060        JoystickButtonWrapper g2b6 = new RobotCode2014.wrappers.JoystickButtonWrapper(gamepad2, 6);
061        JoystickButtonWrapper g2b7 = new RobotCode2014.wrappers.JoystickButtonWrapper(gamepad2, 7);
062        JoystickButtonWrapper g2b8 = new RobotCode2014.wrappers.JoystickButtonWrapper(gamepad2, 8);
063        
064        //Button    8           Collect balls
065        //Button    6           Reject balls
066        g1b6.whilePressed(new CollectCommand());
067        g1b8.whilePressed(new RejectCommand());
068        
069        g1b6.whilePressed(new PitchCommand(-.4));
070        g1b8.whilePressed(new PitchCommand(.4));
071        
072        g1b4.whenPressed(new ToggleClampCommand());
073        
074        //Button    3           Launch at specified networking angle
075        //since both require manipulator, only one gets called 
076        g1b3.whenPressed(new LaunchCommand());
077        
078        g1b1.whenPressed(new UnlockPistonCommand());
079        g1b1.whenReleased(new LockAndRetractPistonCommand());
080        
081        g1b2.toggleWhenPressed(new RetractPistonCommand());
082
083        //b3.whenPressed(new RunWithModifier(b5, new RobotCode2014.commands.manipulator.LaunchBallCommand()));
084        
085        //Button    1 gamepad2  Launch the ball
086        //LaunchBallCommand is medium level code while LaunchCommand is high level code
087        //b9 does the same thing as b12
088        //b9.whenPressed(new LaunchBallCommand());
089        
090        //Button    8 gamepad2  collect ball
091        g2b8.whilePressed(new CollectCommand());
092        
093        //Button    6 gamepad2  Reject ball
094        g2b6.whilePressed(new RejectCommand());
095        
096        //Button    3 gamepad2  Toggle clamp
097        g2b3.whenPressed(new ToggleClampCommand());
098        
099        //BUtton    4 gamepad2  Toggles the piston
100        g2b4.whenPressed(new UnlockPistonCommand());
101        g2b4.whenReleased(new LockPistonCommand());
102
103        SmartDashboard.putData(new LaunchBallCommand());
104        SmartDashboard.putData(new LaunchCommand());
105        
106        SmartDashboard.putData(new TurnToAngleCommand(Math.PI)); //180 degrees
107        SmartDashboard.putData(new PitchToAngleCommand(0));
108        
109        SmartDashboard.putData(new RetractPistonCommand());
110        SmartDashboard.putData(new LockAndRetractPistonCommand());
111        
112        SmartDashboard.putData(new LockPistonCommand());
113        SmartDashboard.putData(new UnlockPistonCommand());
114        
115        SmartDashboard.putData("stop compressor", new SetCompressorState(0));
116        SmartDashboard.putData("start compressor", new SetCompressorState(1));
117        
118        
119    }
120}
